struct Patient {
    int id;             // 挂号单号 (全局唯一)
    int severity;       // 病情危重度 (1-10，数值越大越优先)
    int arrival_time;   // 到达时间戳 (数值越小代表越早到，越优先)
};
auto cmp=[](const Patient& a,const Patient& b){
    if(a.severity!=b.severity){
        return a.severity>b.severity;
    }
    if(a.arrival_time!=b.arrival_time){
        return a.arrival_time<b.arrival_time; 
    }
    return a.id<b.id;
};
set<Patient,decltype(cmp)> patients(cmp);
struct ComparePatient {
    // 重载 "函数调用运算符 ()"
    bool operator()(const Patient& a, const Patient& b) const {
        if(a.severity!=b.severity){
            return a.severity>b.severity;
        }
        if(a.arrival_time!=b.arrival_time){
            return a.arrival_time<b.arrival_time; 
        }
        return a.id<b.id;
    }
};
set<Patient,ComparePatient> patients;
